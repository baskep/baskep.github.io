const leftNav = {
    $newLeftNav: $('.left-nav'),
    $leftNav: $('.left-nav-item'),
    init: function () {
        this.bind();
    },
    bind: function () {
        const self = this;
        self.$leftNav.on('click', function () {
            const _this = this;
            const area = '.' + $(_this).attr('data-area');
            const distance = $(_this).attr('data-area') ? $(area).offset().top : 0;
            $('html, body').animate({
                'scrollTop': distance + 'px'
            }, 1000);
        });
        self.handlePageScroll();
        $(document).scroll(function () {
            self.debounce(self.handlePageScroll, 200);
        });
    },
    // é¡µé¢æ»‘åŠ¨é€‰ä¸­å·¦ä¾§å¯¼èˆªæ 
    handlePageScroll: function () {
        const self = this;
        let scrollEl = $(document.scrollingElement || document.body);
        let top = scrollEl.scrollTop();
        if (top > $('.daily_recom').offset().top) {
            self.$newLeftNav.fadeIn();
        } else {
            self.$newLeftNav.fadeOut();
        }
        let area = null;
        let floor = null;
        let nav = null;
        for (let i = self.$leftNav.length - 1; i >= 0; i--) {
            nav = self.$leftNav.eq(i);
            area = nav.attr('data-area');
            floor = $('.' + area);
            if (!floor || !floor.length) {
                continue;
            } else if (floor.offset().top <= top) {
                nav.addClass('active').siblings().removeClass('active');
                return;
            }
        }
        nav && nav.addClass('active').siblings().removeClass('active');
    },
    // é˜²æŠ–å‡½æ•°
    debounce: function (action, delay) {
        let lastTimer;
        if (lastTimer) {
            return;
        }
        action.apply(this, arguments);
        lastTimer = setTimeout(function () {
            clearTimeout(lastTimer);
            lastTimer = null;
        }, delay);
    }
}
